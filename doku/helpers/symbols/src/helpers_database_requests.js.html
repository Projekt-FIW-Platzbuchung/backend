1 const express = require("express");
  2 const router = express.Router();
  3 const seat = require("./models/seat");
  4 const bookings = require("./models/bookings");
  5 const user = require("./models/user");
  6 
  7 /**
  8  * Aggregates seat with booking collection as bookingDetails and bookingDetails with user-collection, adds status "frei" or "gebucht" to each seat for a date
  9  * @param {Date} date - date to check booking status
 10  * @return {Array} - results of aggregation: seats with properties, status and bookingDetails
 11  */
 12 
 13 async function bookingInformationByDate(date) {
 14   try {
 15     const results = await seat.aggregate([
 16       {
 17         $lookup: {
 18           from: "bookings",
 19           localField: "seatId",
 20           foreignField: "seatId",
 21           as: "bookingDetails",
 22         },
 23       },
 24       {
 25         $addFields: {
 26           bookingDetails: {
 27             $filter: {
 28               input: "$bookingDetails",
 29               as: "booking",
 30               cond: {
 31                 $eq: [{ $toDate: "$$booking.date" }, new Date(date)],
 32               },
 33             },
 34           },
 35         },
 36       },
 37       {
 38         $lookup: {
 39           from: "user",
 40           let: { user_id: { $arrayElemAt: ["$bookingDetails.userId", 0] } },
 41           pipeline: [{ $match: { $expr: { $eq: ["$userId", "$$user_id"] } } }],
 42           as: "userDetails",
 43         },
 44       },
 45       {
 46         $addFields: {
 47           bookingDetails: {
 48             $cond: [
 49               { $eq: [{ $size: "$bookingDetails" }, 0] },
 50               null,
 51               {
 52                 $mergeObjects: [
 53                   { $arrayElemAt: ["$bookingDetails", 0] },
 54                   {
 55                     username: {
 56                       $cond: [
 57                         { $gt: [{ $size: "$userDetails" }, 0] },
 58                         { $arrayElemAt: ["$userDetails.name", 0] },
 59                         null,
 60                       ],
 61                     },
 62                   },
 63                 ],
 64               },
 65             ],
 66           },
 67         },
 68       },
 69       {
 70         $project: {
 71           seatId: 1,
 72           properties: 1,
 73           "bookingDetails.userId": 1,
 74           "bookingDetails.date": 1,
 75           "bookingDetails.username": 1,
 76           status: {
 77             $cond: [{ $ne: ["$bookingDetails", null] }, "gebucht", "frei"],
 78           },
 79           _id: 0,
 80         },
 81       },
 82     ]);
 83 
 84     // console.log(JSON.stringify(results, null, 2));
 85     
 86     return results;
 87   } catch (error) {
 88     console.error("Error during aggregation:", error);
 89     return [];
 90   }
 91 }

 94   bookingInformationByDate,
